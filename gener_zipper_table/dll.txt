CREATE TABLE IF NOT EXISTS ods_trans_contract_task
(
    id                               BIGINT  ,
    merchant_sn                      STRING   COMMENT '商户sn',
    merchant_name                    STRING   COMMENT '商户名称',
    type                             STRING   COMMENT '新增商户入网,基本信息变更,结算账户变更,附件上传,费率变更',
    event_msg                        STRING   COMMENT '处理这个事件的消息内容 可以为空',
    event_context                    STRING   COMMENT '处理这个事件产生的商户上下文参数',
    status                           BIGINT   COMMENT '0待处理 1.处理中 4被终止 5处理成功  6处理失败',
    affect_sub_task_count            BIGINT   COMMENT '影响主任务成功的子任务总数量',
    affect_status_success_task_count BIGINT   COMMENT '已经成功的 影响主任务成功的子任务数量',
    result                           STRING   COMMENT '{"channel":"","message":""} 成功时 目前不需要存 失败时需要存失败的通道信息和原因',
    create_at                        DATETIME,
    update_at                        DATETIME,
    version                          BIGINT   COMMENT '版本号 每次数据有更新+1',
    priority                         DATETIME,
    complete_at                      DATETIME,
    rule_group_id                    STRING   COMMENT '报备规则组'
)
COMMENT '总任务表,分区表,每日增量'
PARTITIONED BY
(
    pt                               STRING
)
LIFECYCLE 36000;
